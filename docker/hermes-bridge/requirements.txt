# Document text extraction (used by agentbuff-multimodal Hermes plugin
# + bridge attachment_preprocessor for /app uploads). One pip install
# covers BOTH the bridge process and the Hermes process — same Python
# env in the container (Dockerfile.hermes installs hermes-agent +
# bridge deps into the same site-packages).
#
# Why each is here:
#   pdfplumber  — preferred PDF text extraction (better layout preservation)
#   pypdf       — fallback if pdfplumber's pdfminer transitive fails to parse
#   python-docx — .docx (Word 2007+) text extraction
#   openpyxl    — .xlsx (Excel 2007+) cell extraction
#   python-pptx — .pptx (PowerPoint 2007+) slide text extraction
#
# These let chief drop ANY common office doc into the agent and get
# its content extracted as text WITHOUT relying on the active chat
# model having native PDF support. Anthropic + Gemini still get
# native passthrough (no extraction); everyone else gets text inline.
pdfplumber==0.11.5
pypdf==5.1.0
python-docx==1.1.2
openpyxl==3.1.5
python-pptx==1.0.2

# Microsoft Edge TTS — kept for compatibility / future upstream fix.
# CURRENTLY BROKEN at the library layer (2026-05-23): Microsoft rotated
# the `TrustedClientToken` + `Sec-MS-GEC` policy and edge-tts 6.1.19's
# hardcoded token now returns `403 Forbidden` at WebSocket handshake.
# Until the upstream library patches this, the AgentBuff multimodal
# plugin demotes `edge` to Tier 99 fallback. Kept installed so if a
# future bump fixes the auth, no rebuild needed — just flip config.
edge-tts==6.1.19

# Google Translate TTS — primary text_to_speech provider after
# edge-tts broke (Bug 4 fix, 2026-05-23). FREE, no API key, no auth,
# supports Indonesian (id-ID) + 50+ other languages, works behind any
# firewall (uses public `translate.google.com` endpoint). Lightweight
# package (~50KB). Registered by the agentbuff-multimodal plugin's
# `tts_providers.py` extension which monkey-patches Hermes' text_to_speech
# tool dispatch table at plugin load (mirror of the 7-provider STT
# extension that already ships in v3.0.0 of the plugin).
gTTS==2.5.4

# WhatsApp per-agent pairing: render Baileys QR string to SVG data URL (server-side)
qrcode>=7.4
